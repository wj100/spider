        let n=arr.length;
        while(n--){
         let index =Math.floor(Math.random()*n);
        [arr[index],arr[n]]=[arr[n],arr[index]]
        }
    }

随机push

在循环中,每次从原数组中剔除一个数放入新数组

function shufflePush(arr){
        var result = [];
        while(arr.length){
            result.push(arr.splice(parseInt(Math.random() * arr.length), 1)[0]);
        }
        return result
    }
